Access Specifiers in C++

Public

Members declared as public are accessible from anywhere in the program. They can be accessed both inside and outside the class.

Example:

class PublicExample {
public:
    int x;
    void display() {
        cout << "Value of x: " << x << endl;
    }
};

int main() {
    PublicExample obj;
    obj.x = 10;
    obj.display();
    return 0;
}
Output:
Value of x: 10
Private

Members declared as private are only accessible within the class. They cannot be accessed directly from outside the class.

Example:

class PrivateExample {
private:
    int x;

public:
    void setX(int value) {
        x = value;
    }
    void display() {
        cout << "Value of x: " << x << endl;
    }
};

int main() {
    PrivateExample obj;
    obj.setX(10);
    obj.display();
    return 0;
}
Output:
Value of x: 10
Protected

Members declared as protected are accessible within the class and by derived classes. They are not accessible from outside the class unless through inheritance.

Example:

class Base {
protected:
    int x;

public:
    void setX(int value) {
        x = value;
    }
};

class Derived : public Base {
public:
    void display() {
        cout << "Value of x: " << x << endl;
    }
};

int main() {
    Derived obj;
    obj.setX(20);
    obj.display();
    return 0;
}
Output:
Value of x: 20
